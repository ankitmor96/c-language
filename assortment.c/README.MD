ðŸ“˜ 1. Largest Element in a 2D Array

=>Description

This program finds and prints the largest element in a 2D array.

=>Source Code

c
#include <stdio.h>

int main() {
    int a, b;

    printf("Enter rows: ");
    scanf("%d", &a);

    printf("Enter columns: ");
    scanf("%d", &b);

    int arr[a][b];

    printf("Enter elements of 2D array:\n");
    for (int i = 0; i < a; i++) {
        for (int j = 0; j < b; j++) {
            scanf("%d", &arr[i][j]);
        }
    }

    int largest = arr[0][0];

    for (int i = 0; i < a; i++) {
        for (int j = 0; j < b; j++) {
            if (arr[i][j] > largest) {
                largest = arr[i][j];
            }
        }
    }

    printf("Largest element = %d", largest);

    return 0;
}

ðŸ“˜ 3. Sum of Each Row & Sum of Each Column (2D Array)

=>Description

This program calculates the row-wise and column-wise sums of a 2D array.

=>Source Code

c
#include <stdio.h>

int main(){
    int a,b;

    printf("Enter the array's row size:");
    scanf("%d", &a);

    printf("Enter the array's column size:");
    scanf("%d", &b);

    int arr[a][b];

    printf("Enter the array elements:\n");

    for(int i=0;i<a;i++){
        for(int j=0;j<b;j++){
            scanf("%d", &arr[i][j]);
        }
    }

    printf("Sum of each row:\n");
    for(int i=0;i<a;i++){
        int rowSum = 0;
        for(int j=0;j<b;j++){
            rowSum += arr[i][j];
        }
        printf("Row %d sum = %d\n", i+1, rowSum);
    }

    printf("Sum of each column:\n");
    for(int i=0;i<b;i++){
        int colSum = 0;
        for(int j=0;j<a;j++){
            colSum += arr[j][i];
        }
        printf("Column %d sum = %d\n", i+1, colSum);
    }

    return 0;
}
